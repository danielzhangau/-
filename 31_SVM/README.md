`test=[6 6; 5.5 5.5; 7 6; 12 14; 7 11; 2 2; 9 9; 8 2; 2 6; 5 10; 4 7; 7 4];` <br>
`groupTest=[1, 1, 1, 2, 2, 2, 2, 2, 2, 2, 1, 1];`<br><br>
